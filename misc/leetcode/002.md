# 2. Add Two Numbers

---

```js
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} l1
 * @param {ListNode} l2
 * @return {ListNode}
 */
var addTwoNumbers = function(l1, l2) {
    var l2n = function(l) {
        var r = 10;
        var n = l.val;
        while (l.next) {
            l = l.next;
            n = n + l.val * r;
            r = r * 10;
        }
        return n;
    };
    var n2l = function(n) {
        if (n < 10) {
            return new ListNode(n);
        } else {
            var val = n % 10;
            n = Math.floor(n / 10);
            var l = new ListNode(val);
            var node = l;
            while (n) {
                val = n % 10;
                n = Math.floor(n / 10);
                node.next = new ListNode(val);
                node = node.next;
            }
            return l;
        }
    };

    var n1 = l2n(l1);
    var n2 = l2n(l2);
    var n3 = n1 + n2;
    var l3 = n2l(n3);
    return l3;
};
```

先把链表转回数字，再将数字转回链表，基本属于暴力解法。

---

当然也可以不要这个过程，在读取列表的时候，直接生成新的列表。
可能还更简单。

```js
var addTwoNumbers = function(l1, l2) {
    var calc = function(l1, l2, prev) {
        var sum = prev;
        if (l1) {
            sum += l1.val;
            l1 = l1.next;
        }
        if (l2) {
            sum += l2.val;
            l2 = l2.next;
        }
        var val, up;
        if (sum < 10) {
            val = sum;
            up = 0;
        } else {
            val = sum % 10;
            up = Math.floor(sum / 10);
        }
        var newNode = new ListNode(val);
        if (l1 || l2 || up) newNode.next = calc(l1, l2, up);
        return newNode;
    };
    return calc(l1, l2, 0);
};
```

偷懒用了递归，换成迭代也不难就是了。
