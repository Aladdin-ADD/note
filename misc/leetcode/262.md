```sql
select
t.Request_at 'Day',
round(count(if(t.Status != 'completed', TRUE, NULL)) / count(*), 2) 'Cancellation Rate'
from Trips t, Users u
where t.Client_Id = u.Users_Id
and u.Role = 'client'
and u.Banned = 'No'
and t.Request_at between '2013-10-01' and '2013-10-03'
group by t.Request_at
```
