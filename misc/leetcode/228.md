# 228. Summary Ranges

---

```js
/**
 * @param {number[]} nums
 * @return {string[]}
 */
var summaryRanges = function(nums) {
    if (nums.length === 0) return nums;
    var start = undefined;
    var prev = undefined;
    var arr = [];
    for (var i = 0, ii = nums.length; i < ii; i++) {
        var curr = nums[i];
        if (start === undefined) {
            start = prev = curr;
        } else {
            if (prev === curr - 1) {
                prev = curr;
            } else {
                if (start === prev) {
                    arr.push(String(start));
                } else {
                    arr.push(String(start) + '->' + String(prev));
                }
                start = prev = curr;
            }
        }
    }
    if (start === prev) {
        arr.push(String(start));
    } else {
        arr.push(String(start) + '->' + String(prev));
    }
    return arr;
};
```
