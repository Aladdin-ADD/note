# 65. Valid Number

```c
bool isNumber(char* s) {
	bool dot_used = false;
	bool e_used = false;
	bool first = true;

	while (*s == ' ') s++;
	if (*s == '\0') return false;

	while (*s != '\0') {
		if (*s == ' ') break;

		if (first) {
			if (*s == '-' || *s == '+') {
				s++;
			}
			if (*s == 'e') {
				return false;
			} else if (*s == '.' && !(*(s+1) <= '9' && *(s+1) >= '0')) {
				return false;
			} else {
				first = false;
			}
		}

		if (*s == '.') {
			if (dot_used || e_used) {
				return false;
			} else {
				dot_used = true;
				s++;
				continue;
			}
		}

		if (*s == 'e') {
			if (e_used) {
				return false;
			}
			if (*(s+1) == '+' || *(s+1) == '-') {
				s++;
			}
			if (!(*(s+1) <= '9' && *(s+1) >= '0')) {
				return false;
			} else {
				e_used = true;
				s++;
				continue;
			}
		}

		if (*s >= '0' && *s <= '9') {
			s++;
			continue;
		}

		return false;
	}
	while (*s == ' ') s++;
	return (*s == '\0' ? true : false);
}
```
