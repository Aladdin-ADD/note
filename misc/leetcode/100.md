# 100. Same Tree

---

```c
bool isSameTree(struct TreeNode* p, struct TreeNode* q) {
	if (p == NULL && q == NULL) {
		return true;
	} else if (p != NULL && q != NULL) {
		if (p->val != q->val) return false;
		bool left = isSameTree(p->left, q->left);
		bool right = isSameTree(p->right, q->right);
		return left && right;
	} else {
		return false;
	}
};
```

```js
var isSameTree = function(p, q) {
    if (p === null && q === null) return true;
    if (p === null || q === null) return false;

    var pArr = [p];
    var qArr = [q];
    while (pArr.length && qArr.length) {
        var pp = pArr.pop();
        var qq = qArr.pop();
        if (pp.val !== qq.val) return false;

        if (pp.right !== null) pArr.push(pp.right);
        if (qq.right !== null) qArr.push(qq.right);
        if (pArr.length !== qArr.length) return false;

        if (pp.left !== null) pArr.push(pp.left);
        if (qq.left !== null) qArr.push(qq.left);
        if (pArr.length !== qArr.length) return false;
    }

    return true;
};
```
