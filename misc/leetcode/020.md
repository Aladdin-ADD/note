# 20. Valid Parentheses

---

```
bool isValid(char* s) {
	int pArr[strlen(s)];
	int idx = 0;
	while (*s != '\0') {
		if (*s == '(') {
			pArr[++idx] = 1;
		} else if (*s == '{') {
			pArr[++idx] = 2;
		} else if (*s == '[') {
			pArr[++idx] = 3;
		} else if (*s == ')') {
			if (pArr[idx--] != 1) return false;
		} else if (*s == '}') {
			if (pArr[idx--] != 2) return false;
		} else if (*s == ']') {
			if (pArr[idx--] != 3) return false;
		}
		s++;
	}
	return (idx == 0 ? true : false);
}
```
