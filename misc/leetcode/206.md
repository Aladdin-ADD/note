# 206. Reverse Linked List

---

```c
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
struct ListNode* reverseList(struct ListNode* head) {
    struct ListNode *tail = NULL;
    struct ListNode *curr = NULL;

    while (head != NULL) {
        curr = tail;
        tail = head;
        head = tail->next;
        tail->next = curr;
    }

    return tail;
}
```
