# 26. Remove Duplicates from Sorted Array

---

```c
int removeDuplicates(int* nums, int numsSize) {
    if (numsSize < 2) return numsSize;
    if (numsSize == 2) return (nums[0] == nums[1] ? 1 : 2);

    int i = 0;
    int j = 1;
    int k = 2;

    while (k < numsSize) {
        if (nums[i] < nums[j]) {
            i++;
            j++;
            k++;
        } else if (nums[k] > nums[i]) {
            nums[j] = nums[k];
            i++;
            j++;
            k++;
        } else {
            k++;
        }
    }

    if (nums[i] < nums[j]) j++;

    return j;
}
```

```c
int removeDuplicates(int* nums, int numsSize) {
    if (numsSize < 2) return numsSize;
    int i = 0;
    for (int j = 1; j < numsSize; j++) {
        if (nums[i] < nums[j]) {
            i++;
            nums[i] = nums[j];
        }
    }
    return i + 1;
}
```
