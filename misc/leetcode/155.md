# 155. Min Stack

---

如果是通用的，应该做不到不遍历就取到最小值吧？
但是结构可以自定义，在这个特殊场景下，可行做法之一是 struct 里直接存储目前为止的最小值。

如果要通用一些，我能想到的是实现一个最小堆，插入删除的时候更新一下堆。

---

```c
typedef struct ms {
    int val;
    int min;
    struct ms *next;
} MinStack;

void minStackCreate(MinStack *stack, int maxSize) {
    stack = malloc(sizeof(MinStack));
    stack->next = NULL;
}

void minStackPush(MinStack *stack, int element) {
    MinStack *push = malloc(sizeof(MinStack));
    push->val = element;
    if (stack->next != NULL) {
        push->next = stack->next;
        push->min = (stack->next->min > push->val ? push->val : stack->next->min);
    } else {
        push->next = NULL;
        push->min = push->val;
    }
    stack->next = push;
}

void minStackPop(MinStack *stack) {
    MinStack *pop = stack->next;
    if (pop != NULL) stack->next = pop->next;
    free(pop);
}

int minStackTop(MinStack *stack) {
    return stack->next->val;
}

int minStackGetMin(MinStack *stack) {
    return stack->next->min;
}

void minStackDestroy(MinStack *stack) {
    MinStack *next;
    while (stack->next != NULL) {
        next = stack->next;
        stack->next = next->next;
        free(next);
    }
    free(stack);
}
```

```js
/**
 * @constructor
 */
var MinStack = function() {
    this.stack = [];
    this.min = [];
};

/**
 * @param {number} x
 * @returns {void}
 */
MinStack.prototype.push = function(x) {
    this.stack.unshift(x);
    var prev = this.min[0];
    this.min.unshift(prev < x ? prev : x);
};

/**
 * @returns {void}
 */
MinStack.prototype.pop = function() {
    this.stack.shift();
    this.min.shift();
};

/**
 * @returns {number}
 */
MinStack.prototype.top = function() {
    return this.stack[0];
};

/**
 * @returns {number}
 */
MinStack.prototype.getMin = function() {
    return this.min[0];
};
```
